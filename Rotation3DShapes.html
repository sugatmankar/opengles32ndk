And Finally we are ready to render , update display method with proper calls
    glUseProgram(gShaderProgramObject);

    glm::mat4 modelMatrix = glm::mat4(1.0);
    glm::mat4 viewMatrix = glm::mat4(1.0);
    glm::mat4 translateMatrix = glm::mat4(1.0);
    glm::mat4 rotationMatrix = glm::mat4(1.0);

    translateMatrix = glm::translate(glm::mat4(1.0),glm::vec3(-1.5f,0.0f,-4.0f));
    rotationMatrix = glm::rotate(glm::mat4(1),glm::radians(gAngle),glm::vec3(0.0f,1.0f,0.0f));
    modelMatrix *= translateMatrix;
    modelMatrix *= rotationMatrix;

    glUniformMatrix4fv(gMUniform,1,GL_FALSE,glm::value_ptr(modelMatrix));
    glUniformMatrix4fv(gVUniform,1,GL_FALSE,glm::value_ptr(viewMatrix));
    glUniformMatrix4fv(gPUniform,1,GL_FALSE,glm::value_ptr(gPerspectiveProjectionMatrix));

    glBindVertexArray(gVao_pyramid);
    glDrawArrays(GL_TRIANGLES,0,12);
    glBindVertexArray(0);

    //cube
    modelMatrix = glm::mat4(1.0);
    viewMatrix = glm::mat4(1.0);
    translateMatrix = glm::mat4(1.0);
    rotationMatrix = glm::mat4(1.0);

    translateMatrix = glm::translate(glm::mat4(1.0),glm::vec3(1.5f,0.0f,-4.0f));
    rotationMatrix = glm::rotate(glm::mat4(1),glm::radians(gAngle),glm::vec3(1.0f,1.0f,1.0f));
    modelMatrix *= translateMatrix;
    modelMatrix *= rotationMatrix;

    glUniformMatrix4fv(gMUniform,1,GL_FALSE,glm::value_ptr(modelMatrix));
    glUniformMatrix4fv(gVUniform,1,GL_FALSE,glm::value_ptr(viewMatrix));
    glUniformMatrix4fv(gPUniform,1,GL_FALSE,glm::value_ptr(gPerspectiveProjectionMatrix));

    glBindVertexArray(gVao_cube);

    glDrawArrays(GL_TRIANGLE_FAN, 0, 4);
    glDrawArrays(GL_TRIANGLE_FAN, 4, 4);
    glDrawArrays(GL_TRIANGLE_FAN, 8, 4);
    glDrawArrays(GL_TRIANGLE_FAN, 12, 4);
    glDrawArrays(GL_TRIANGLE_FAN, 16, 4);
    glDrawArrays(GL_TRIANGLE_FAN, 20, 4);

    glBindVertexArray(0);

    glUseProgram(0);

    update();
